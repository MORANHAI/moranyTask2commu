קישור רכיבי UI לקוד Java (`MainActivity.java`)

כדי שקוד ה-Java יוכל לתקשר עם רכיבי ה-UI שיצרנו ב-XML, עלינו "למצוא" אותם באמצעות ה-`id` הייחודי שלהם.

קובץ XML (`activity_main.xml`)
<EditText
    android:id="@+id/celsiusInput"
    android:inputType="numberDecimal"
    ...
/>

<Button
    android:id="@+id/convertButton"
    android:text="המר לפרנהייט"
    ...
/>

<TextView
    android:id="@+id/fahrenheitResult"
    android:text="תוצאה:"
    ...
/>
➡️
קוד Java (`MainActivity.java`)
public class MainActivity extends AppCompatActivity {

    EditText celsiusInput;
    Button convertButton;
    TextView fahrenheitResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // מוצאים את הרכיבים לפי ה-ID שלהם
        celsiusInput = findViewById(R.id.celsiusInput);
        convertButton = findViewById(R.id.convertButton);
        fahrenheitResult = findViewById(R.id.fahrenheitResult);
    }
}

קבלת קלט והמרת נתונים

קבלת טקסט מ-EditText:


String celsiusString = celsiusInput.getText().toString();
// חשוב! getText() מחזיר Editable, לכן חובה להוסיף toString()

המרת טקסט למספר (Double):

חשוב: קלט ממשתמש הוא תמיד טקסט! חייבים להמיר אותו למספר כדי לבצע חישובים. 


double celsius = Double.parseDouble(celsiusString);
// Double.parseDouble() ממיר String למספר מסוג double.

טיפול בשגיאות (מיד נרחיב): מה קורה אם המשתמש מזין "abc" במקום מספר?

יישום הלוגיקה החישובית

את הלוגיקה נמקם בתוך ה"מאזין" (Listener) של הכפתור, שיופעל בלחיצה.

נוסחת המרה ממעלות צלזיוס לפרנהייט:

$$F = C \times \frac{9}{5} + 32$$ 


import android.view.View; // חובה להוסיף ייבוא זה

// בתוך MainActivity.java, בתוך שיטת onCreate() לאחר איתור הרכיבים
convertButton.setOnClickListener(new View.OnClickListener() {
    @Override
    public void onClick(View v) {
        // שלב 1: קבלת קלט המשתמש
        String celsiusString = celsiusInput.getText().toString();

        // בדיקה אם השדה ריק (טיפול שגיאה ראשוני)
        if (celsiusString.isEmpty()) {
            fahrenheitResult.setText("שגיאה: אנא הכנס טמפרטורה.");
            return; // עצור את פעולת הלחצן כאן
        }

        try {
            // שלב 2: המרת קלט טקסטואלי למספר
            double celsius = Double.parseDouble(celsiusString);

            // שלב 3: ביצוע החישוב
            double fahrenheit = celsius * 9.0 / 5.0 + 32.0;

            // שלב 4: הצגת התוצאה למשתמש (נרחיב בשקף הבא)
            // fahrenheitResult.setText("הטמפרטורה בפרנהייט: " + fahrenheit);

        } catch (NumberFormatException e) {
            // טיפול בשגיאה אם הקלט אינו מספר (נרחיב בשקף נפרד)
            fahrenheitResult.setText("שגיאה: אנא הכנס מספר חוקי.");
        }
    }
});

הצגת פלט ב-`TextView`

לאחר שביצענו את החישוב, אנו רוצים להציג את התוצאה למשתמש.

קוד Java (`MainActivity.java`)
// בתוך ה-onClick() של הכפתור:

double celsius = /* ... קלט מהמשתמש ... */;
double fahrenheit = celsius * 9.0 / 5.0 + 32.0;

// 1. הצגת התוצאה באמצעות setText()
fahrenheitResult.setText("הטמפרטורה בפרנהייט: " + fahrenheit);

// 2. עיצוב הפלט (מומלץ)
String formattedResult = String.format("%.2f", fahrenheit);
fahrenheitResult.setText("הטמפרטורה בפרנהייט: " + formattedResult);

// הפונקציה String.format() מאפשרת לעצב
// מחרוזות בצורה גמישה, לדוגמה: 2 ספרות אחרי הנקודה.

טיפול בשגיאות קלט (`Error Handling`)

אפליקציה יציבה וידידותית למשתמש חייבת לטפל במצבי שגיאה, כמו קלט לא חוקי.

ממיר טמפרטורה
המר לפרנהייט
שגיאה: אנא הכנס מספר חוקי.
קוד Java (`MainActivity.java`)
// בתוך ה-onClick() של הכפתור:

String celsiusString = celsiusInput.getText().toString();

// 1. בדיקה אם השדה ריק
if (celsiusString.isEmpty()) {
    fahrenheitResult.setText("שגיאה: אנא הכנס טמפרטורה.");
    return;
}

try {
    // 2. ניסיון להמיר ולחשב
    double celsius = Double.parseDouble(celsiusString);
    double fahrenheit = celsius * 9.0 / 5.0 + 32.0;
    fahrenheitResult.setText("הטמפרטורה בפרנהייט: " + String.format("%.2f", fahrenheit));

} catch (NumberFormatException e) {
    // 3. טיפול בשגיאת המרה (קלט שאינו מספר)
    fahrenheitResult.setText("שגיאה: אנא הכנס מספר חוקי (לדוגמה: 25.5).");
}

בחרו אחת מהאפשרויות הבאות:

אפשרות 1: ממיר טמפרטורה דו-כיווני

  • הרחיבו את אפליקציית ממיר הצלזיוס-פרנהייט.
  • הוסיפו כפתור נוסף: "המר לצלזיוס".
  • יישמו את הלוגיקה ההפוכה: $$C = (F - 32) \times \frac{5}{9}$$
  • ודאו שאתם משתמשים באותו EditText לקלט, אך התוצאה מוצגת ב-TextView המתאים (או שאלו כיצד לשנות את הכותרת של ה-TextView).
  • **חובה:** ודאו שטיפול השגיאות עובד לשני הכיוונים!

אפשרות 2: ממיר יחידות חדש - מטרים לרגל

  • צרו אפליקציה חדשה הממירה מטרים לרגל ($1 \text{ מטר} = 3.28084 \text{ רגל}$).
  • השתמשו ב-EditText לקלט, Button להמרה ו-TextView לפלט.

חובה: לטפל בשגיאות קלט!

טיפ: חשבו על ההיגיון שעומד מאחורי כל המרה, ואיך הייתם פותרים זאת ב-Java לפני שהייתם בונים את ה-UI.

סיכום

היום בנינו גשר בין ידע ה-Java שלכם לעולם האפליקציות.

  • למדנו על רכיבי UI אינטראקטיביים (`EditText`, `Button`, `TextView`).
  • הבנו כיצד לקבל קלט, לבצע חישוב, ולהציג פלט.
  • למדנו על החשיבות של טיפול בשגיאות קלט (`try-catch`).
